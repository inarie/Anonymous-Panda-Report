\chapter*{List of Acronyms}
% Define a convenient command to add a line
% to the database
\newcommand*{\addacronym}[2]{%
  \DTLnewrow{acronyms}%
  \DTLnewdbentry{acronyms}{Acronym}{#1}%
  \DTLnewdbentry{acronyms}{Description}{#2}%
}


% Create the database
\DTLnewdb{acronyms}
\addacronym{UE4}{Unreal Engine 4}
\addacronym{VBO}{Virtual Body Ownership}
\addacronym{VR}{Virtual Reality}
\addacronym{Vtubers}{Virtual Youtubers}
\addacronym{AFI}{Anticipated Future Interaction}
\addacronym{DDC}{Display Data Channel}
\addacronym{XR}{Extended Reality}
\addacronym{VH}{Virtual Human(s)}
\addacronym{VC}{Virtual Character(s)}
\addacronym{VA}{Virtual Avatar(s)}
\addacronym{LOD}{Level of Detail}
\addacronym{AE}{Affective Empathy}
\addacronym{CE}{Cognitive Empathy}
\addacronym{SP}{Social Presence}
\addacronym{HMD}{Head Mounted Display}
\addacronym{DMD}{Desktop Mounted Display}
\addacronym{CMD}{Chest Mounted Display}


% Sort the database
\DTLsort{Acronym}{acronyms}

% Display the contents of the database
\begin{itemize}
\DTLforeach*{acronyms}{\thisAcronym=Acronym,\thisDesc=Description}%
  {\item[] \textbf{\thisAcronym} \thisDesc}%
\end{itemize}
